cmake_minimum_required(VERSION 2.8)

project(kdtree)

	${kdtree_SOURCE_DIR}
	${kdtree_SOURCE_DIR}/src
	${EIGEN2_INCLUDE_DIR})

add_definitions(-O2 -ggdb)

add_executable(kdtreeTest test/main.cpp)
target_link_libraries(kdtreeTest cpptest)
